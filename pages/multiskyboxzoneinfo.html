class MultiSkyZoneInfo expands ZoneInfo;

simulated function LinkToSkybox()
{
    local skyzoneinfo Sky;

    Super.LinkToSkybox();

    foreach AllActors( class 'SkyZoneInfo', Sky, Event )
    {
        SkyZone = Sky;
    }
}
